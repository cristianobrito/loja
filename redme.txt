  /* Reseta as margens e paddings do HTML e body, define altura total da tela e aplica box-sizing para incluir padding e border no tamanho do elemento */
html, body {
  margin:  0;             /* remove margens padrão do navegador */
  padding: 0;             /* remove padding padrão do navegador */
  height: 100%;           /* define altura total da janela      */
  box-sizing: border-box; /* inclui borda e padding no cálculo do tamanho do elemento */
}

/* Container principal do dashboard */
.dashboard-container{
    display: grid;                      /* define que será um grid container */
    grid-template-columns: 1fr 6fr 3fr; /* 3 colunas: 1 fr para menu, 6 fr para conteúdo, 3 fr para painel direito */
    grid-template-areas: "a m p";       /* define nomes das áreas do grid: aside (a), main (m), painel direito (p) */
    gap: 20px;                          /* espaçamento entre colunas e linhas */
    height: 100vh;                      /* altura do container igual a altura da viewport (tela) */
    overflow: hidden;                   /* evita scroll, esconde excesso de conteúdo */
    margin: 0px 50px 0px 25px;          /* margem externa do dashboard: topo 5px, direita 50px, baixo 5px, esquerda 25px */
}

/* Estilo do menu lateral (aside) */
aside{
    /* background-color: chartreuse; */
    background-color: #f1f1f1;  /* cor de fundo do menu lateral */
    grid-area: a;                 /* define que ocupa a área 'a' no grid principal */
    display: grid;                /* transforma o aside em grid para posicionar itens */
    grid-template-rows: repeat(8, 1fr);/* divide o aside em 8 linhas iguais */
    padding: 10px 0;                   /* espaçamento interno no topo e baixo */
    border-radius: 25px;               /* cantos arredondados do menu lateral */
}

/* Estilo dos quadradinhos de cada ícone */
.ico-personal{
    display: grid;                         /* transforma cada item em grid para poder centralizar ícone e label */
    color: coral;                        /* cor inicial do conteúdo (texto e ícone) */
    cursor: pointer;                       /* muda o cursor para mãozinha ao passar por cima */
    transition: transform 0.5s, color 0.5s;/* animação suave para transformação e mudança de cor */
    place-items: center;                   /* centraliza ícone e texto */
    row-gap: 5px;                          /* espaço entre linhas dentro do grid (ícone e label) */
    color: inherit;                        /* herda cor do pai */
    background-color: rgba(255, 255, 255, 0.2);/* fundo branco semi-transparente */
    border: 1px solid rgba(255, 255, 255, 0.3);/* borda branca semi-transparente */
    border-radius: 10px;                         /* cantos arredondados */
    padding: 10px;                               /* espaçamento interno */
    box-shadow: 0 4px 6px rgba(0, 0, 0, 0.2);  /* sombra leve embaixo do quadrado */
}


/* Efeito ao passar o mouse sobre os quadradinhos */
.ico-personal:hover {
  color: coral;        /* muda a cor        */
  transform: scale(1.2); /* aumenta levemente */
}

/* Estilo do link dentro do quadrado */
.ico-personal a {
  display: grid;          /* mantém grid dentro do link */
  place-items: center;    /* centraliza ícone + label */
  text-decoration: none;  /* remove underline */
  /* color: inherit;*/
  color: rgb(181, 173, 170);  /* cor do ícone e label dentro do link */
}

/* Hover do link dentro do quadrado */
.ico-personal a:hover {
  text-decoration: none;  /* garante que não apareça sublinhado */
  color: inherit;         /* herda cor do quadradinho */

como exibir os dados no dashboard

<div id="dadosBTC"></div>

<script>
fetch("https://www.mercadobitcoin.net/api/BTC/ticker/")
  .then(response => response.json())
  .then(json => {
    const t = json.ticker;
    document.getElementById("dadosBTC").innerHTML = `
      <h2>Bitcoin (BRL)</h2>
      <p>💰 Último preço: R$ ${t.last}</p>
      <p>⬆️ Alta do dia: R$ ${t.high}</p>
      <p>⬇️ Baixa do dia: R$ ${t.low}</p>
      <p>📈 Volume: ${t.vol}</p>
    `;
  });
</script>

Percentuais	Donut ou gauge
Um número principal	Card grande / KPI
